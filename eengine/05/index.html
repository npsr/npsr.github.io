        class App extends React.Component {
          constructor(props) {
            super(props);
            this.state = {
              counter: 0,
              items: [1, 2]
            };
          }
          // Bindowanie
          increase = () => {
            // setState (nie el.state)
            this.setState(state => ({
              counter: state.counter + 1
            }));
          };
          // JW
          addItem = () => {
            // JW
            this.setState(state => ({
              items: [...state.items, state.items.length + 1]
            }));
          };
          render() {
            return ( <
              React.Fragment >
              <
              div className = "data" >
              <
              div > {
                this.state.counter
              } < /div> <
              ul > {
                this.state.items.map(item => ( <
                  li key = {
                    item
                  } > element {
                    item
                  } < /li>
                ))
              } <
              /ul> < /
              div > <
              div className = "actions" >
              <
              button type = "button"
              onClick = {
                this.increase
              } >
              zwiększ <
              /button> <
              button type = "button"
              onClick = {
                this.addItem
              } >
              dodaj <
              /button> < /
              div > <
              /React.Fragment>
            );
          }
        }
